package org.yscheon.image

import android.content.Context
import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.EditText
import android.widget.Toast
import kotlinx.android.synthetic.main.activity_image.*

class Image : AppCompatActivity() {


    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_image)

        var listner1 = btsListner()
        button.setOnClickListener(listner1)
    }
    inner class btsListner: View.OnClickListener{
        override fun onClick(v: View?) {
            if(editText.text.toString()=="Blueberry"){
                imageview.setVisibility(View.VISIBLE)
                imageview.setImageResource(R.drawable.blueberry)
            }
            else if(editText.text.toString()=="Peach"){
                imageview.setVisibility(View.VISIBLE)
                imageview.setImageResource(R.drawable.peach)
            }
            else if(editText.text.toString()=="Tomato"){
                imageview.setVisibility(View.VISIBLE)
                imageview.setImageResource(R.drawable.tomato)
            }
            else{
                imageview.setVisibility(View.INVISIBLE)
            }
        }
    }
}
